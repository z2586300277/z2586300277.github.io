const material = new THREE.SpriteMaterial({
    map: texture,
    transparent: true,
    depthWrite: false,  // йҖҸжҳҺеҸ еҠ ж—¶дёҚеҶҷж·ұеәҰ

const sprite = new THREE.Sprite(material);

и„үеҶІеҠЁз”»йҖ»иҫ‘ вҖӢ

js
let circle_n = 0;
function animate() {
    circle_n = circle_n > 10 ? 0 : circle_n + 0.2;
    const scale = 0.1 * circle_n + 0.1;
    sprite.scale.set(scale, scale, scale);
    sprite.material.opacity = Math.max(1 - circle_n * 0.05, 0.2);
}

circle_n 0вҶ’10 еҫӘзҺҜпјҡscale д»Һе°ҸеҲ°еӨ§пјҢopacity д»Һ 1 еҲ° 0.2пјҢз„¶еҗҺйҮҚзҪ®гҖӮ
